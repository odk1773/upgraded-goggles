    <div class="photo-container">
        <img id="targetImage" src="" alt="사진을 불러오는 중...">
    </div>

    <script>
        // 1. 주소창에서 ?file= 뒤에 오는 사진 파일명을 슥 읽어옵니다.
        const urlParams = new URLSearchParams(window.location.search);
        const fileName = urlParams.get('file');
        
        // 2. 만약 주소창에 파일명이 존재한다면 사진을 매핑합니다.
        if (fileName) {
            // ★ 깃허브 Pages 안전 경로: 같은 방(루트)에 있는 사진을 다이렉트로 읽도록 보정했습니다.
            document.getElementById('targetImage').src = "./" + fileName;
        } else {
            // 주소창에 ?file=이 없을 때 뜨는 예외 처리 안내문
            document.getElementById('targetImage').alt = "주소창에 사진 파일명이 지정되지 않았습니다.";
        }
    </script>
